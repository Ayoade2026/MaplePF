        function addRow(tableId, desc='', owner='Joint', val=0) {

            const row = tbody.insertRow();

                <td class="p-3">

                    <select class="w-full bg-transparent focus:outline-none" onchange="syncTableToState()">

                        <option value="Individual" ${owner==='Individual'?'selected':''}>Individual</option>

                        <option value="Spouse" ${owner==='Spouse'?'selected':''}>Spouse</option>

        function populateTablesFromState() {

            document.getElementById('asset-table').getElementsByTagName('tbody')[0].innerHTML = '';

        function syncTableToState() {

        }


        // --- SUB-ROUTINE: Financial Computation Engines ---

        function recalculateConsolidatedBalances() {

            let totalAssets = appState.assets.reduce((sum, a) => sum + a.val, 0);

            let totalLiabs = appState.liabilities.reduce((sum, l) => sum + l.val, 0);

            let netWorth = totalAssets - totalLiabs;


            document.getElementById('dash-networth').innerText = new Intl.NumberFormat('en-CA', { style: 'currency', currency: 'CAD', maximumFractionDigits: 0 }).format(netWorth);

            calculateFIRE(netWorth);

        }


        function calculateFIRE(currentNetWorth) {

            if (currentNetWorth === undefined) {

                let totalAssets = appState.assets.reduce((sum, a) => sum + a.val, 0);

                let totalLiabs = appState.liabilities.reduce((sum, l) => sum + l.val, 0);

                currentNetWorth = totalAssets - totalLiabs;

            }

            const annualExpenses = parseFloat(document.getElementById('fire-expenses').value) || 0;

            const swr = parseFloat(document.getElementById('fire-swr').value) || 4.0;

            

            const targetFund = annualExpenses / (swr / 100);

            const progress = targetFund > 0 ? Math.min((currentNetWorth / targetFund) * 100, 100) : 0;


            document.getElementById('dash-fire-target').innerText = new Intl.NumberFormat('en-CA', { style: 'currency', currency: 'CAD', maximumFractionDigits: 0 }).format(targetFund);

            document.getElementById('dash-fire-progress').innerText = `${progress.toFixed(1)}%`;

        }

        function runCompoundCalc() {

            const p = parseFloat(document.getElementById('comp-principal').value) || 0;

            const m = parseFloat(document.getElementById('comp-monthly').value) || 0;

            const r = (parseFloat(document.getElementById('comp-rate').value) || 0) / 100 / 12;

            const y = parseFloat(document.getElementById('comp-years').value) || 0;

            

            let total = p;

            for(let i = 0; i < y * 12; i++) {

                total = total * (1 + r) + m;

            }

            document.getElementById('comp-result').innerText = new Intl.NumberFormat('en-CA', { style: 'currency', currency: 'CAD', maximumFractionDigits: 0 }).format(total);

        }


        function runDebtCalc() {

            const bal = parseFloat(document.getElementById('debt-amount').value) || 0;

            const apr = (parseFloat(document.getElementById('debt-rate').value) || 0) / 100 / 12;

            const pmt = parseFloat(document.getElementById('debt-payment').value) || 0;


            if (pmt <= bal * apr) {

                document.getElementById('debt-result').innerText = "Payment too low (Interest accumulating)";

                return;

            }

            let months = Math.log(pmt / (pmt - bal * apr)) / Math.log(1 + apr);

            document.getElementById('debt-result').innerText = isNaN(months) ? "N/A" : `${Math.ceil(months)} Months`;

        }


        function runTaxCalc() {

            const income = parseFloat(document.getElementById('tax-income').value) || 0;

            const prov = document.getElementById('tax-province').value;

            

            // Simplified Combined 2026 Marginal Tax Estimates (Federal + Provincial structural approximation)

            const taxBrackets = {

                AB: [ { threshold: 55000, rate: 0.25 }, { threshold: 110000, rate: 0.32 }, { threshold: 165000, rate: 0.38 }, { threshold: Infinity, rate: 0.44 } ],

                ON: [ { threshold: 52000, rate: 0.20 }, { threshold: 105000, rate: 0.30 }, { threshold: 150000, rate: 0.38 }, { threshold: Infinity, rate: 0.47 } ],

                BC: [ { threshold: 47000, rate: 0.20 }, { threshold: 95000, rate: 0.28 }, { threshold: 145000, rate: 0.36 }, { threshold: Infinity, rate: 0.45 } ]

            };

            

            const brackets = taxBrackets[prov] || taxBrackets['ON']; // Fallback calculation structure

            let taxOwed = 0;

            let previousThreshold = 0;


            for (let bracket of brackets) {

                if (income > bracket.threshold) {

                    taxOwed += (bracket.threshold - previousThreshold) * bracket.rate;

                    previousThreshold = bracket.threshold;

                } else {

                    taxOwed += (income - previousThreshold) * bracket.rate;

                    break;

                }

            }

            

            const effRate = income > 0 ? (taxOwed / income) * 100 : 0;

            document.getElementById('tax-owed').innerText = new Intl.NumberFormat('en-CA', { style: 'currency', currency: 'CAD', maximumFractionDigits: 0 }).format(taxOwed);

            document.getElementById('tax-rate').innerText = `${effRate.toFixed(1)}%`;

        }

        // --- SUB-ROUTINE: Local Native Cryptography Pipeline ---

        async function deriveKey(password, salt) {

            const enc = new TextEncoder();

            const baseKey = await crypto.subtle.importKey("raw", enc.encode(password), "PBKDF2", false, ["deriveKey"]);

            return crypto.subtle.deriveKey(

                { name: "PBKDF2", salt: salt, iterations: 100000, hash: "SHA-256" },

                baseKey,

                { name: "AES-GCM", length: 256 },

                false,

                ["encrypt", "decrypt"]

            );

        }


        async function initVault() {

            const password = document.getElementById('master-password').value;

            if(!password) return;

            const errorLabel = document.getElementById('auth-error');

            errorLabel.classList.add('hidden');


            try {

                const encryptedData = localStorage.getItem('canwealth_vault');

                if (encryptedData) {

                    const payload = JSON.parse(encryptedData);

                    const salt = new Uint8Array(payload.salt);

                    const iv = new Uint8Array(payload.iv);

                    cryptoKey = await deriveKey(password, salt);

                    

                    const decrypted = await crypto.subtle.decrypt(

                        { name: "AES-GCM", iv: iv },

                        cryptoKey,

                        new Uint8Array(payload.ciphertext)

                    );

                    appState = JSON.parse(new TextDecoder().decode(decrypted));

                } else {

                    // Initialize empty clean vault profiles

                    const salt = crypto.getRandomValues(new Uint8Array(16));

                    cryptoKey = await deriveKey(password, salt);

                    // Force inject a fresh current record matching timeline parameters

                    appState.trendHistory.push({ date: 'Today', networth: 399000 });

                }

                

                document.getElementById('auth-overlay').classList.add('hidden');

                populateTablesFromState();

                runInitialCalculations();

            } catch (e) {

                errorLabel.classList.remove('hidden');

                console.error(e);

            }

        }


        async function saveAllData() {

            if(!cryptoKey) return alert("Vault not unlocked.");

            syncTableToState();

            

            const enc = new TextEncoder();

            const iv = crypto.getRandomValues(new Uint8Array(12));

            const salt = crypto.getRandomValues(new Uint8Array(16));

            

            // Re-derive key with salt to store alongside package structure safely

            const rawPassword = document.getElementById('master-password').value;

            const operationalKey = await deriveKey(rawPassword, salt);


            const ciphertext = await crypto.subtle.encrypt(

                { name: "AES-GCM", iv: iv },

                operationalKey,

                enc.encode(JSON.stringify(appState))

            );


            const payload = {

                ciphertext: Array.from(new Uint8Array(ciphertext)),

                iv: Array.from(iv),

                salt: Array.from(salt)

            };


            localStorage.setItem('canwealth_vault', JSON.stringify(payload));

            alert("Data Vault encrypted and saved securely to LocalStorage.");

        }
